for i in {2010..2017}
  do 
    mkdir year_$i
      for j in {1..12}
        do
           if [ $j -lt 10 ]
	     then
	       mkdir year_$i/month_0$j
               echo 'Файл 001' > year_$i/month_0$j/001.txt
               echo 'Файл 002' > year_$i/month_0$j/002.txt
             else
               mkdir year_$i/month_$j
               echo 'Файл 001' > year_$i/month_$j/001.txt
               echo 'Файл 002' > year_$i/month_$j/002.txt
           fi
        done
done
